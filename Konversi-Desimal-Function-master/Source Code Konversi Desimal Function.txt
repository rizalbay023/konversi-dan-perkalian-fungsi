void db (int des) 
{
	string simpan;
	if (des == 0)
	{
		cout << "0";
	}
	else if (des == 1)
	{
		cout << "1";
	}
	else
	{
		for (int i = 0; i != des; des /= 2)
		{
			int hasil = des % 2;
			string convert = to_string(hasil);
			simpan.append(convert);
		}
		int panjang = simpan.length();
		for (int i = panjang; i != 0; i--)
		{
			cout << simpan[i - 1];
		}
	}
}

void dh (int des) 
{
	string simpan;
	const int desTetap = des;
	for (int i = 0; des != i; des /= 16)
	{
		int bil = des % 16;
		string convert;
		switch (bil)
		{
		case 10:
			convert = "A";
			break;
		case 11:
			convert = "B";
			break;
		case 12:
			convert = "C";
			break;
		case 13:
			convert = "D";
			break;
		case 14:
			convert = "E";
			break;
		case 15:
			convert = "F";
			break;
		default:
			convert = to_string(bil);
		}

		simpan.append(convert);
	}
	int length = simpan.length();
	if (desTetap == 0) {
		cout << "0";
	}
	for (int i = length; i != 0; i--)
	{
		cout << simpan[i - 1];
	}
}
